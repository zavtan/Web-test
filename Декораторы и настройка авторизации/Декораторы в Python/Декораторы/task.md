```python
import time

def timing_decorator(func):
    def wrapper(*args, **kwargs):  # –¢–∞–∫ –∫–∞–∫ –∑–∞—Ä–∞–Ω–µ–µ –Ω–µ –∏–∑–≤–µ—Å—Ç–Ω–æ —Å –∫–∞–∫–∏–º–∏ –∞—Ä–≥—É–º–µ–Ω—Ç–∞–º–∏ –±—É–¥–µ—Ç –¥–µ–∫–æ—Ä–∏—Ä–æ–≤–∞—Ç—å—Å—è —Ñ—É–Ω–∫—Ü–∏—è –∏—Å–ø–æ–ª—å–∑—É–µ–º *args, **kwargs
        start_time = time.time()  # –ó–∞–ø–æ–º–∏–Ω–∞–µ–º —Ç–µ–∫—É—â–µ–µ –≤—Ä–µ–º—è –ø–µ—Ä–µ–¥ –≤—ã–∑–æ–≤–æ–º —Ñ—É–Ω–∫—Ü–∏–∏
        result = func(*args, **kwargs)  # –í—ã–∑—ã–≤–∞–µ–º –æ—Ä–∏–≥–∏–Ω–∞–ª—å–Ω—É—é —Ñ—É–Ω–∫—Ü–∏—é
        end_time = time.time()  # –ó–∞–ø–æ–º–∏–Ω–∞–µ–º —Ç–µ–∫—É—â–µ–µ –≤—Ä–µ–º—è –ø–æ—Å–ª–µ –≤—ã–ø–æ–ª–Ω–µ–Ω–∏—è —Ñ—É–Ω–∫—Ü–∏–∏
        elapsed_time = end_time - start_time  # –í—ã—á–∏—Å–ª—è–µ–º —Ä–∞–∑–Ω–∏—Ü—É –≤—Ä–µ–º–µ–Ω–∏
        print(f"–í—Ä–µ–º—è —Ä–∞–±–æ—Ç—ã {func.__name__}: {elapsed_time} —Å–µ–∫—É–Ω–¥")  # func.__name__ –ø–æ–∑–≤–æ–ª—è–µ—Ç –ø–æ–ª—É—á–∏—Ç—å –∏–º—è —Ñ—É–Ω–∫—Ü–∏–∏
        return result  # –í–æ–∑–≤—Ä–∞—â–∞–µ–º —Ä–µ–∑—É–ª—å—Ç–∞—Ç –≤—ã–ø–æ–ª–Ω–µ–Ω–∏—è –æ—Ä–∏–≥–∏–Ω–∞–ª—å–Ω–æ–π —Ñ—É–Ω–∫—Ü–∏–∏
    return wrapper  # –í–æ–∑–≤—Ä–∞—â–∞–µ–º —Ñ—É–Ω–∫—Ü–∏—é-–æ–±–µ—Ä—Ç–∫—É

@timing_decorator
def function_1(arg1, arg2):
    # –ù–µ–∫–æ—Ç–æ—Ä—ã–π –∫–æ–¥ —Ñ—É–Ω–∫—Ü–∏–∏
    time.sleep(2)
    return arg1 + arg2

@timing_decorator
def function_2(arg1, arg2, arg3):
    # –ù–µ–∫–æ—Ç–æ—Ä—ã–π –∫–æ–¥ —Ñ—É–Ω–∫—Ü–∏–∏
    time.sleep(3)
    return arg1 * arg2 * arg3

# –í—ã–∑–æ–≤ —Ñ—É–Ω–∫—Ü–∏–π —Å –¥–µ–∫–æ—Ä–∞—Ç–æ—Ä–æ–º
result_1 = function_1(3, 4)
result_2 = function_2(2, 5, 6)

```
